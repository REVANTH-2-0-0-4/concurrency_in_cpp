#include<iostream>
#include<future>     // this future  class provides a machanism to access  the results of the async operation 
using namespace std;
int square(int x){
	return x*x;
}

	future<int> asyncfunciton = async(&async,12);
	for(int i = 0 ; i < 10 ; i++){
		cout<<square(i)<<endl;
	}
